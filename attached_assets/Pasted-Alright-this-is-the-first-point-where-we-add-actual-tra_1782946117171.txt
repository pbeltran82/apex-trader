This is a FastAPI + alpaca-py paper order endpoint.


✅ COPY / PASTE:

/api/paper-order

Add this under your existing endpoints in main.py:

from fastapi import Request

# -----------------------------
# SAFE PAPER ORDER ENDPOINT
# -----------------------------
@app.post("/api/paper-order")
async def paper_order(request: Request):
    try:
        data = await request.json()

        symbol = data.get("symbol")
        qty = float(data.get("qty", 0))
        side = data.get("side")  # "buy" or "sell"
        order_type = data.get("type", "market")

        # -----------------------------
        # SAFETY RULES (IMPORTANT)
        # -----------------------------

        if not symbol or side not in ["buy", "sell"]:
            return {"error": "Invalid symbol or side"}

        if qty <= 0:
            return {"error": "Quantity must be greater than 0"}

        # HARD LIMIT: prevent large accidental orders
        if qty > 10:
            return {"error": "Safety limit: max 10 shares per order"}

        # -----------------------------
        # CREATE ORDER
        # -----------------------------
        order = client.submit_order(
            symbol=symbol.upper(),
            qty=qty,
            side=side,
            type=order_type,
            time_in_force="day"
        )

        return {
            "id": order.id,
            "status": order.status,
            "symbol": order.symbol,
            "qty": float(order.qty),
            "side": order.side,
            "type": order.type
        }

    except Exception as e:
        return {"error": str(e)}
